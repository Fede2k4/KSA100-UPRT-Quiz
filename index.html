        let currentQuestionIndex = 0;

        let score = 0;

        let studentName = "";

        let timerInterval;

        let timeLeft = 0;

        let studentAnswers = [];


        function showScreen(screenId) {

            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

            document.getElementById(screenId).classList.add('active');

        }


        function startQuiz() {

            studentName = document.getElementById('student-name').value.trim();

            if (!studentName) {

                alert("Per favore, inserisci il tuo nome.");

                return;

            }

            showScreen('screen-quiz');

            loadQuestion();

        }


        function loadQuestion() {

            clearInterval(timerInterval);

            if (currentQuestionIndex >= quizData.length) {

                endQuiz();

                return;

            }


            const currentQuestion = quizData[currentQuestionIndex];

            document.getElementById('question-text').innerText = `${currentQuestionIndex + 1}. ${currentQuestion.question}`;

            

            const imgElement = document.getElementById('quiz-img');

            if (currentQuestion.image && currentQuestion.image !== "") {

                imgElement.src = currentQuestion.image;

                imgElement.style.display = "inline-block";

            } else {

                imgElement.style.display = "none";

            }


            const optionsContainer = document.getElementById('options-container');

            optionsContainer.innerHTML = '';

            

            currentQuestion.options.forEach((option, index) => {

                const btn = document.createElement('button');

                btn.className = 'option-btn';

                btn.innerText = option;

                btn.onclick = () => selectAnswer(index);

                optionsContainer.appendChild(btn);

            });


            timeLeft = currentQuestion.time * 1000;

            updateTimerBar();

            

            const tickRate = 100; 

            timerInterval = setInterval(() => {

                timeLeft -= tickRate;

                updateTimerBar();

                if (timeLeft <= 0) {

                    selectAnswer(-1); 

                }

            }, tickRate);

        }


        function updateTimerBar() {

            const totalTime = quizData[currentQuestionIndex].time * 1000;

            const percentage = Math.max(0, (timeLeft / totalTime) * 100);

            document.getElementById('timer').style.width = percentage + '%';

        }


        function selectAnswer(selectedIndex) {

            clearInterval(timerInterval);

            const currentQuestion = quizData[currentQuestionIndex];

            const isCorrect = selectedIndex === currentQuestion.correct;

            

            if (isCorrect) score++;


            studentAnswers.push({

                question: currentQuestion.question,

                studentAnswer: selectedIndex === -1 ? "Tempo Scaduto" : currentQuestion.options[selectedIndex],

                correctAnswer: currentQuestion.options[currentQuestion.correct],

                status: isCorrect ? "CORRETTA" : "ERRATA"

            });


            currentQuestionIndex++;

            loadQuestion();

        }


        function endQuiz() {

            showScreen('screen-results');

            document.getElementById('final-score').innerText = `Ottimo lavoro ${studentName}! Punteggio finale: ${score} / ${quizData.length}`;

            

            let resultString = `STUDENTE: ${studentName}\nDATA DI SVOLGIMENTO: ${new Date().toLocaleString()}\nPUNTEGGIO GENERALE: ${score}/${quizData.length}\n\nRELAZIONE DI DETTAGLIO:\n`;

            studentAnswers.forEach((ans, idx) => {

                resultString += `${idx+1}) ${ans.question}\n   Risposta data dallo studente: ${ans.studentAnswer} [${ans.status}]\n`;

            });


            document.getElementById('results-code').innerText = resultString;

        }
